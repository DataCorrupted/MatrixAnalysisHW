	\item[Problem 2] (Problem Description)
	.\\
	\textbf{i) U}
	$$\forall \lambda \in \sigma(U), |\lambda| = 1$$
	\begin{proof}
		Suppose we have eigenpair $(\lambda, v)$:
		$$ Uv = \lambda v$$
		$$ v^*U^* = v^*\lambda^*$$
		$$ v^*U^* Uv = v^*v = v^*\lambda^*\lambda v$$
		$$ v^*v = \lambda^*\lambda v^*v$$
		$$ \lambda^*\lambda = |\lambda| = 1$$
	\end{proof}
